Panasonic Let's note SV1 Linux风扇问题解决

前言

最近入手了一款外观复古的松下机器。想着把linux装在上面，但只要一进linux系统，风扇便开始狂转，吵得难以使用下去。

一开始以为是驱动的问题，但手动编译了多次内核依然如故，于是便在网上另辟蹊径。经过一番苦苦搜索，最终在这里发现了解决问题的思路：

  #!/bin/shell
  c(){ echo "$1" | tee /proc/acpi/call >/dev/null && cat /proc/acpi/call;echo;}
  c '\_SB.PCI0.LPCB.EC0.EC88 0xB5 0x79 0x05 0x00'
  c '\_SB.PCI0.LPCB.EC0.TFN1._FSL 0'

原文作者说在他的QV9和SR4上面 0xB5 0x79 0x05 0x00 能将风扇设置为静音模式， _FSL 能手动设置风扇速度（上述PCI0在我的SV1上需要换成PC00）。

不过可惜的是， 发送 0xB5 0x79 0x05 0x00 后，电脑直接关闭风扇了。又再折腾几次，发觉该指令在我机器上反倒类似于获取风扇控制权，无奈只好往手动控制风扇上开拓思路。

藉由AI的帮助，用rust勉强完成了一个还算能用的风扇守护程序：

  use std::fs::{self, File};
  use std::io;
  use std::os::unix::fs::FileExt;
  use std::path::{Path, PathBuf};
  use std::sync::atomic::{AtomicBool, Ordering};
  use std::time::{Duration, Instant};

  const ACPI:     &str     = "/proc/acpi/call";
  const EC_ON:    &[u8]    = b"\\_SB.PC00.LPCB.EC0.EC88 0xB5 0x79 0x05 0x00";
  const EC_OFF:   &[u8]    = b"\\_SB.PC00.LPCB.EC0.EC88 0xB5 0x79 0x00 0x00";
  const COOLDOWN: Duration = Duration::from_secs(2);
  const RETRY:    Duration = Duration::from_secs(1);

  struct Tier { up: i32, dn: i32, cmd: &'static [u8], poll: Duration }

  const TIERS: &[Tier] = &[
      Tier { up: 100, dn: 77, cmd: b"\\_SB.PC00.LPCB.EC0.TFN1._FSL 100", poll: Duration::from_millis(300)  },
      Tier { up: 85,  dn: 67, cmd: b"\\_SB.PC00.LPCB.EC0.TFN1._FSL 90",  poll: Duration::from_millis(500)  },
      Tier { up: 75,  dn: 59, cmd: b"\\_SB.PC00.LPCB.EC0.TFN1._FSL 75",  poll: Duration::from_millis(800)  },
      Tier { up: 68,  dn: 53, cmd: b"\\_SB.PC00.LPCB.EC0.TFN1._FSL 62",  poll: Duration::from_millis(1000) },
      Tier { up: 62,  dn: 48, cmd: b"\\_SB.PC00.LPCB.EC0.TFN1._FSL 55",  poll: Duration::from_millis(1000) },
      Tier { up: 57,  dn: 41, cmd: b"\\_SB.PC00.LPCB.EC0.TFN1._FSL 50",  poll: Duration::from_millis(1000) },
      Tier { up: 50,  dn: 34, cmd: b"\\_SB.PC00.LPCB.EC0.TFN1._FSL 45",  poll: Duration::from_millis(1000) },
      Tier { up: 43,  dn: 0,  cmd: b"\\_SB.PC00.LPCB.EC0.TFN1._FSL 40",  poll: Duration::from_millis(1000) },
  ];

  static RUNNING: AtomicBool = AtomicBool::new(true);

  extern "C" fn stop(_: libc::c_int) {
      RUNNING.store(false, Ordering::Relaxed);
  }

  struct Devices {
      acpi:   File,
      temp:   File,
      smooth: Option<i32>,
  }

  impl Devices {
      // 构造即代表"EC_ON 已成功发出"，Drop 发送 EC_OFF 的前提由此保证
      fn open(tp: &Path, tier_cmd: &[u8]) -> Option<Self> {
          let acpi = File::options().write(true).open(ACPI).ok()?;
          acpi.write_at(EC_ON,    0).ok()?;
          acpi.write_at(tier_cmd, 0).ok()?;
          Some(Self {
              acpi,
              temp:   File::open(tp).ok()?,
              smooth: None,
          })
      }

      fn write_acpi(&self, cmd: &[u8]) -> io::Result<()> {
          self.acpi.write_at(cmd, 0).map(|_| ())
      }

      fn read_temp(&self) -> Option<i32> {
          let mut buf = [0u8; 16];
          let n = self.temp.read_at(&mut buf, 0).ok()?;
          std::str::from_utf8(&buf[..n]).ok()?
              .trim()
              .parse::<i32>().ok()
              .map(|v| v / 1000)
      }
  }

  // Devices 存在 ⟺ EC_ON 已激活，生命周期与资源激活周期精确对应
  impl Drop for Devices {
      fn drop(&mut self) {
          let _ = self.acpi.write_at(EC_OFF, 0);
      }
  }

  fn find_temp_path() -> PathBuf {
      fs::read_dir("/sys/class/thermal")
          .expect("Cannot read /sys/class/thermal")
          .filter_map(|e| e.ok())
          .find_map(|e| {
              let mut p = e.path();
              p.push("type");
              let ok = fs::read_to_string(&p).is_ok_and(|t| t.contains("pkg_temp"));
              p.set_file_name("temp");
              ok.then_some(p)
          })
          .expect("No pkg_temp thermal zone")
  }

  fn decide(lvl: usize, s: i32, elapsed: Duration) -> usize {
      if lvl > 0 && s >= TIERS[lvl].up {
          return (0..lvl).rev().find(|&i| s < TIERS[i].up).unwrap_or(0);
      }
      if s < TIERS[lvl].dn && elapsed > COOLDOWN {
          return (lvl..TIERS.len()).find(|&i| s >= TIERS[i].dn).unwrap();
      }
      lvl
  }

  fn tick(d: &mut Devices, lvl: usize, elapsed: Duration) -> Option<usize> {
      let temp = d.read_temp()?;
      let s = d.smooth.get_or_insert(temp);
      ,*s = if temp > *s { (temp + *s + 1) >> 1 } else { (temp + 7 * *s + 4) >> 3 };
      let new = decide(lvl, *s, elapsed);
      if new != lvl && d.write_acpi(TIERS[new].cmd).is_err() { return None; }
      Some(new)
  }

  fn main() {
      if unsafe { libc::geteuid() } != 0 {
          eprintln!("Root required");
          std::process::exit(1);
      }

      unsafe {
          let mut sa: libc::sigaction = std::mem::zeroed();
          sa.sa_sigaction = stop as *const () as libc::sighandler_t;
          libc::sigaction(libc::SIGINT,  &sa, std::ptr::null_mut());
          libc::sigaction(libc::SIGTERM, &sa, std::ptr::null_mut());
      }

      let tp = find_temp_path();

      let mut devices: Option<Devices> = None;
      let mut lvl  = TIERS.len() - 1;
      let mut last = Instant::now() - COOLDOWN;

      while RUNNING.load(Ordering::Relaxed) {

          // ── 长效句柄：仅在断开时重建 ─────────────────────────────────────
          if devices.is_none() {
              match Devices::open(&tp, TIERS[lvl].cmd) {
                  None    => { std::thread::sleep(RETRY); continue; }
                  Some(d) => { devices = Some(d); }
              }
          }

          // ── 采样、决策、写入 ─────────────────────────────────────────────
          let now = Instant::now();
          let d = devices.as_mut().expect("devices guaranteed Some after setup");
          match tick(d, lvl, now.duration_since(last)) {
              Some(new) => {
                  if new != lvl { lvl = new; last = now; }
              }
              None => { devices = None; continue; }
          }

          std::thread::sleep(TIERS[lvl].poll);
      }

      // devices 在此 drop，Drop impl 自动发送 EC_OFF
  }

具体步骤

  1. 安装acpi_call内核模块，确定 lsmod | grep acpi_call 会输出acpi_call

  2. 编译rust代码，个人所用的Cargo.toml为：

      [package]
  name = "fan-controld"
  version = "0.1.0"
  edition = "2021"
  authors = ["Your Name"]
  description = "ACPI fan control daemon"

  [dependencies]
  libc = "0.2"

  [profile.release]
  opt-level = "z"

  lto = true

  codegen-units = 1

  strip = true

  panic = "abort"

想要静态链接，可选用musl

3.开机自启，以及唤醒后重新启动 我的系统是slackware，启动时所用脚本为：

  #!/bin/sh
  #
  # /etc/rc.d/rc.fan-controld
  # Initialization script for the custom ACPI fan control daemon.

  DAEMON="/usr/local/sbin/fan-controld"
  PIDFILE="/var/run/fan-controld.pid"

  fan_start() {
    if [ -f "$PIDFILE" ] && kill -0 $(cat "$PIDFILE") 2>/dev/null;
    then
      echo "fan-controld is already running (PID: $(cat $PIDFILE))."
      return
    fi

    if [ ! -x "$DAEMON" ];
    then
      echo "Error: Daemon binary not found or not executable at $DAEMON."
      exit 1
    fi

    echo "Starting fan-controld..."
    nohup "$DAEMON" > /dev/null 2>&1 &

    echo $! > "$PIDFILE"
  }

  fan_stop() {
    if [ ! -f "$PIDFILE" ];
    then
      echo "fan-controld is not running (no PID file)."
      return
    fi

    PID=$(cat "$PIDFILE")

    if kill -0 "$PID" 2>/dev/null;
    then
      echo "Stopping fan-controld (PID: $PID)..."
      kill -TERM "$PID"
      sleep 1
    else
      echo "fan-controld is not running, cleaning up stale PID file."
    fi

    rm -f "$PIDFILE"
  }

  fan_status() {
    if [ -f "$PIDFILE" ] && kill -0 $(cat "$PIDFILE") 2>/dev/null;
    then
      echo "fan-controld is running (PID: $(cat $PIDFILE))."
    else
      echo "fan-controld is stopped."
    fi
  }

  case "$1" in
    'start')
      fan_start
      ;;
    'stop')
      fan_stop
      ;;
    'restart')
      fan_stop
      sleep 1
      fan_start
      ;;
    'status')
      fan_status
      ;;
    ,*)
      echo "Usage: $0 {start|stop|restart|status}"
      exit 1
      ;;
  esac

由于睡眠后会重置风扇控制权，我直接为elogind加了一个钩子：

  #!/bin/sh

  # /lib64/elogind/system-sleep/fan-control.sh
  # Hook to restart fan-controld during suspend/resume

  if [ -x /etc/rc.d/rc.fan-controld ]; then
        /etc/rc.d/rc.fan-controld restart
  fi
