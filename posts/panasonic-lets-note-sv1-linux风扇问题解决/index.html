Panasonic Let's note SV1 Linux风扇问题解决

前言

最近入手了一款外观复古的松下机器。想着把linux装在上面，但只要一进linux系统，风扇便开始狂转，吵得难以使用下去。

一开始以为是驱动的问题，但手动编译了多次内核依然如故，于是便在网上另辟蹊径。经过一番苦苦搜索，最终在这里发现了解决问题的思路：

  #!/bin/shell
  c(){ echo "$1" | tee /proc/acpi/call >/dev/null && cat /proc/acpi/call;echo;}
  c '\_SB.PCI0.LPCB.EC0.EC88 0xB5 0x79 0x05 0x00'
  c '\_SB.PCI0.LPCB.EC0.TFN1._FSL 0'

原文作者说在他的QV9和SR4上面 0xB5 0x79 0x05 0x00 能将风扇设置为静音模式， _FSL 能手动设置风扇速度（上述PCI0在我的SV1上需要换成PC00）。

不过可惜的是， 发送 0xB5 0x79 0x05 0x00 后，电脑直接关闭风扇了。又再折腾几次，发觉该指令在我机器上反倒类似于获取风扇控制权，无奈只好往手动控制风扇上开拓思路。

藉由AI的帮助，用rust勉强完成了一个还算能用的风扇守护程序：

  use std::fs::{self, File};
  use std::io::Write;
  use std::os::unix::fs::FileExt;
  use std::path::{Path, PathBuf};
  use std::process::ExitCode;
  use std::sync::atomic::{AtomicBool, Ordering};
  use std::time::{Duration, Instant};

  const ACPI:     &str     = "/proc/acpi/call";
  const EC_ON:    &[u8]    = b"\\_SB.PC00.LPCB.EC0.EC88 0xB5 0x79 0x05 0x00";
  const EC_OFF:   &[u8]    = b"\\_SB.PC00.LPCB.EC0.EC88 0xB5 0x79 0x00 0x00";

  // ── 调优参数 ──────────────────────────────────────────────────────────────────
  const COOLDOWN: Duration = Duration::from_secs(5);
  const RETRY:    Duration = Duration::from_secs(1);

  // ── 档位表 ───────────────────────────────────────────────────────────────────
  // 字段说明：
  //   up  — 温度超过此值时升到更高档（立即响应，无冷却）
  //   dn  — 温度低于此值且经过 COOLDOWN 后才降到更低档
  //   cmd — 发给 EC 的风扇速度命令（百分比）
  //   poll— 此档位下的采样间隔
  #[derive(Clone, Copy, Debug)]
  struct Tier { up: i32, dn: i32, cmd: &'static [u8], poll: Duration }

  const TIERS: &[Tier] = &[
      // tier 0：≥88°C，全速，快速采样
      Tier { up: 100, dn: 78, cmd: b"\\_SB.PC00.LPCB.EC0.TFN1._FSL 100", poll: Duration::from_millis(300) },
      // tier 1：≥80°C
      Tier { up: 88,  dn: 72, cmd: b"\\_SB.PC00.LPCB.EC0.TFN1._FSL 90",  poll: Duration::from_millis(400) },
      // tier 2：≥74°C
      Tier { up: 80,  dn: 65, cmd: b"\\_SB.PC00.LPCB.EC0.TFN1._FSL 78",  poll: Duration::from_millis(600) },
      // tier 3：≥68°C
      Tier { up: 74,  dn: 58, cmd: b"\\_SB.PC00.LPCB.EC0.TFN1._FSL 66",  poll: Duration::from_millis(800) },
      // tier 4：≥62°C
      Tier { up: 68,  dn: 52, cmd: b"\\_SB.PC00.LPCB.EC0.TFN1._FSL 55",  poll: Duration::from_millis(1000) },
      // tier 5：≥56°C
      Tier { up: 62,  dn: 46, cmd: b"\\_SB.PC00.LPCB.EC0.TFN1._FSL 45",  poll: Duration::from_millis(1500) },
      // tier 6：≥50°C
      Tier { up: 56,  dn: 40, cmd: b"\\_SB.PC00.LPCB.EC0.TFN1._FSL 40",  poll: Duration::from_millis(2000) },
      // tier 7：<50°C，待机档
      // dn = 0 是必须的：保证 decide() 的降档搜索总能命中此档兜底
      Tier { up: 50,  dn: 0,  cmd: b"\\_SB.PC00.LPCB.EC0.TFN1._FSL 35",  poll: Duration::from_millis(2000) },
  ];

  const LAST_TIER: usize = TIERS.len() - 1;

  // 编译期验证 TIERS 不变量
  const _: () = {
      assert!(!TIERS.is_empty(), "TIERS 不能为空");          // ← 新增：防御空切片
      assert!(TIERS[LAST_TIER].dn == 0, "最低档 dn 必须为 0");
      let mut i = 0;
      while i < TIERS.len() {
          assert!(TIERS[i].up > TIERS[i].dn, "up 必须大于同档 dn（滞回区间为正）");
          if i + 1 < TIERS.len() {
              assert!(TIERS[i].up > TIERS[i + 1].up, "up 阈值必须严格递减");
              assert!(TIERS[i].dn > TIERS[i + 1].dn, "dn 阈值必须严格递减");
          }
          i += 1;
      }
  };

  static RUNNING: AtomicBool = AtomicBool::new(true);

  extern "C" fn stop(_: libc::c_int) {
      RUNNING.store(false, Ordering::Release);
  }

  // ── RAII Guard ────────────────────────────────────────────────────────────────
  // Devices::open 初始化期间，EC_ON 已激活但 Devices 尚未构造。
  // 此 Guard 确保任何失败路径（包括 ? 短路）都自动发送 EC_OFF。
  // 初始化全部成功后调用 forget() 解除，所有权移交 Devices::drop。
  struct EcGuard<'a>(&'a File);

  impl Drop for EcGuard<'_> {
      #[cold]
      fn drop(&mut self) {
          let _ = (&*self.0).write_all(EC_OFF);
      }
  }

  struct Devices {
      acpi:   File,
      temp:   File,
      smooth: i32,  // EMA 平滑温度
      trend:  i32,  // 温度变化速率的平滑估计（Holt 趋势项）
  }

  impl Devices {
      #[cold]
      #[must_use]
      fn open(tier_cmd: &[u8], temp_path: &Path) -> Option<Self> {
          let acpi = File::options().write(true).open(ACPI).ok()?;

          (&acpi).write_all(EC_ON).ok()?;

          // EC_ON 成功后激活 Guard，任何 ? 失败均自动发送 EC_OFF
          let guard = EcGuard(&acpi);
          (&acpi).write_all(tier_cmd).ok()?;          // ← 同上，用 write_all

          let temp = File::open(temp_path).ok()?;
          let init = read_temp_file(&temp)?;

          std::mem::forget(guard); // 初始化成功，移交给 Devices::drop
          Some(Self { acpi, temp, smooth: init, trend: 0 })
      }

      #[must_use]
      #[inline]
      fn write_acpi(&self, cmd: &[u8]) -> Option<()> {
          (&self.acpi).write_all(cmd).ok()
      }
  }

  // sysfs 温度文件格式：纯 ASCII 十进制整数（单位 m°C）+ 换行。
  // 直接解析字节，跳过 from_utf8 + trim + parse 三步，减少热路径开销。
  // 缓冲区 8 字节足够：最大合理值约 150000（6 位）+ 换行 = 7 字节。
  #[inline]
  fn read_temp_file(f: &File) -> Option<i32> {
      let mut buf = [0u8; 8];
      let n = f.read_at(&mut buf, 0).ok()?;
      let end = buf[..n].iter().position(|b| !b.is_ascii_digit()).unwrap_or(n);
      if end == 0 { return None; }
      // checked_mul/add：把溢出纳入 Option 错误路径，而非 panic 或静默截断
      let millideg = buf[..end].iter().try_fold(0i32, |a, &b| {
          a.checked_mul(10)?.checked_add((b - b'0') as i32)
      })?;
      Some(millideg / 1000)
  }

  // Devices 存在 ⟺ EC_ON 已激活，生命周期与资源激活周期精确对应
  impl Drop for Devices {
      #[cold]
      fn drop(&mut self) {
          let _ = (&self.acpi).write_all(EC_OFF);
      }
  }

  fn init_temp_path() -> Result<PathBuf, &'static str> {
      fs::read_dir("/sys/class/thermal")
          .map_err(|_| "Cannot read /sys/class/thermal")?
          .filter_map(|e| e.ok())
          .find_map(|e| {
              let p = e.path();
              fs::read_to_string(p.join("type"))
                  // ← 改进：trim() 消除尾部换行后再 contains，
                  // 比直接 contains 更精确，避免误匹配含 "pkg_temp" 的其他字串。
                  // 使用 contains 而非 == "x86_pkg_temp"，保持对不同架构的兼容性
                  // （AMD 与 Intel 均使用 x86_pkg_temp，ARM/RISC-V 不同）。
                  .is_ok_and(|t| t.trim().contains("pkg_temp"))
                  .then(|| p.join("temp"))
          })
          .ok_or("No pkg_temp thermal zone")
  }

  #[inline]
  fn decide(lvl: usize, s: i32, elapsed: Duration) -> usize {
      // 升档：立即响应，无冷却
      // TIERS[..lvl].partition_point(|t| t.up > s) 找到第一个 up ≤ s 的位置，
      // 减 1 即为"仍能容纳当前温度（up > s）的最低优先级档位"，即目标档
      if lvl > 0 && s >= TIERS[lvl].up {
          return TIERS[..lvl].partition_point(|t| t.up > s).saturating_sub(1);
      }
      // 降档：需满足冷却时间，防止在两档之间频繁抖动
      // TIERS[LAST_TIER].dn == 0 已由编译期断言保证，partition_point 必然命中
      if s < TIERS[lvl].dn && elapsed > COOLDOWN {
          return TIERS[lvl..].partition_point(|t| t.dn > s) + lvl;
      }
      lvl
  }

  #[must_use]
  #[inline]
  fn tick(d: &mut Devices, lvl: usize, elapsed: Duration) -> Option<usize> {
      let temp = read_temp_file(&d.temp)?;

      // ── Holt 双重指数平滑 ────────────────────────────────────────────────────
      // 第一层：非对称 EMA 平滑原始温度
      //   升温 α=0.5（快速追踪，避免高负载响应滞后）
      //   降温 α≈0.06（缓慢跟随，抑制短暂回落触发频繁降档）
      //
      // 注：i32::midpoint（1.85 稳定）向零截断，与此处升温需要的向正无穷取整语义不同，
      // 故不替换：(temp + smooth + 1) >> 1 保持"偏保守（偏高）"的升温追踪行为。
      let prev = d.smooth;
      d.smooth = if temp > d.smooth {
          (temp + d.smooth + 1) >> 1   // 升温：四舍五入（向正无穷），快速追踪
      } else {
          (temp + 15 * d.smooth) >> 4  // 降温：截断（保守方向），α≈0.0625
      };

      // 第二层：EMA 平滑温度变化速率（趋势项），β≈0.125
      //   delta > 0：温度上升中；delta < 0：温度下降中
      let delta = d.smooth - prev;
      // 用 / 而非 >>：>> 对负数向负无穷截断，导致负趋势永远无法衰减到 0；
      // / 向零截断，trend=-1,delta=0 → (-7)/8=0，负趋势可正常衰减
      d.trend = (delta + 7 * d.trend) / 8;

      // ── 有效温度：smooth + 限幅后的趋势投影 ────────────────────────────────
      // trend > 0（升温趋势）：有效温度适当前移，风扇提前响应
      // trend < 0（降温趋势）：有效温度略微前移，降档更果断
      // clamp 防止趋势项失控：最多投影 +8°C / -3°C
      let effective = d.smooth + d.trend.clamp(-3, 8);

      let new = decide(lvl, effective, elapsed);
      if new != lvl { d.write_acpi(TIERS[new].cmd)?; }
      Some(new)
  }

  fn setup_signals() {
      unsafe {
          let mut sa: libc::sigaction = std::mem::zeroed();
          sa.sa_sigaction = stop as *const () as libc::sighandler_t;
          // SA_RESTART：确保 Drop 中的 EC_OFF write_all 在信号连续到达时仍能完整发出；
          // 次要作用：防止控制循环中的 read_at/write_all 被 EINTR 误判为设备失联
          sa.sa_flags = libc::SA_RESTART;
          for &sig in &[libc::SIGINT, libc::SIGTERM] {
              libc::sigaction(sig, &sa, std::ptr::null_mut());
          }
      }
  }

  // ExitCode 替代 process::exit()：通过正常返回路径运行析构，
  // 确保 Devices::drop 在退出时也能发送 EC_OFF
  fn run(temp_path: &Path) -> ExitCode {
      let mut lvl  = LAST_TIER;
      // checked_sub：系统启动不足 COOLDOWN 秒时安全兜底
      let mut last = Instant::now().checked_sub(COOLDOWN).unwrap_or_else(Instant::now);

      // 双层循环分离职责：
      //   外层 'outer — 连接管理，负责 Devices 的构造与断线重连
      //   内层        — 控制循环，devices 在此永远是具体值，无需 Option 或 unsafe
      'outer: loop {
          if !RUNNING.load(Ordering::Acquire) { break; }

          // ── 重连直到成功或收到停止信号 ───────────────────────────────────────
          let mut devices = loop {
              if !RUNNING.load(Ordering::Acquire) { break 'outer; }
              if let Some(d) = Devices::open(TIERS[lvl].cmd, temp_path) { break d; }
              std::thread::sleep(RETRY);
          };

          // ── 控制循环：采样、决策、写入 ───────────────────────────────────────
          while RUNNING.load(Ordering::Acquire) {
              let now = Instant::now();

              // let..else：设备失联时跳出内层，外层负责重连
              let Some(new) = tick(&mut devices, lvl, now.duration_since(last)) else {
                  break;
              };
              if new != lvl { lvl = new; last = now; }

              std::thread::sleep(TIERS[lvl].poll);
          }
          // devices 在此 drop，Drop impl 自动发送 EC_OFF
      }

      ExitCode::SUCCESS
  }

  fn main() -> ExitCode {
      if unsafe { libc::geteuid() } != 0 {
          eprintln!("Root required");
          return ExitCode::FAILURE;
      }

      setup_signals();

      let path = match init_temp_path() {
          Ok(p)  => p,
          Err(e) => { eprintln!("{e}"); return ExitCode::FAILURE; }
      };

      run(&path)
  }

具体步骤

  1. 安装acpi_call内核模块，确定 lsmod | grep acpi_call 会输出acpi_call

  2. 编译rust代码，个人所用的Cargo.toml为：

      [package]
  name = "fan-controld"
  version = "0.1.0"
  edition = "2024"

  [dependencies]

想要静态链接，可选用musl

3.开机自启，以及唤醒后重新启动 我的系统是slackware，启动时所用脚本为：

  #!/bin/sh
  #
  # /etc/rc.d/rc.fan-controld
  # Initialization script for the custom ACPI fan control daemon.

  DAEMON="/usr/local/sbin/fan-controld"
  PIDFILE="/var/run/fan-controld.pid"

  fan_start() {
    if [ -f "$PIDFILE" ] && kill -0 $(cat "$PIDFILE") 2>/dev/null;
    then
      echo "fan-controld is already running (PID: $(cat $PIDFILE))."
      return
    fi

    if [ ! -x "$DAEMON" ];
    then
      echo "Error: Daemon binary not found or not executable at $DAEMON."
      exit 1
    fi

    echo "Starting fan-controld..."
    nohup "$DAEMON" > /dev/null 2>&1 &

    echo $! > "$PIDFILE"
  }

  fan_stop() {
    if [ ! -f "$PIDFILE" ];
    then
      echo "fan-controld is not running (no PID file)."
      return
    fi

    PID=$(cat "$PIDFILE")

    if kill -0 "$PID" 2>/dev/null;
    then
      echo "Stopping fan-controld (PID: $PID)..."
      kill -TERM "$PID"
      sleep 1
    else
      echo "fan-controld is not running, cleaning up stale PID file."
    fi

    rm -f "$PIDFILE"
  }

  fan_status() {
    if [ -f "$PIDFILE" ] && kill -0 $(cat "$PIDFILE") 2>/dev/null;
    then
      echo "fan-controld is running (PID: $(cat $PIDFILE))."
    else
      echo "fan-controld is stopped."
    fi
  }

  case "$1" in
    'start')
      fan_start
      ;;
    'stop')
      fan_stop
      ;;
    'restart')
      fan_stop
      sleep 1
      fan_start
      ;;
    'status')
      fan_status
      ;;
    ,*)
      echo "Usage: $0 {start|stop|restart|status}"
      exit 1
      ;;
  esac
